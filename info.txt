git init - инициализация пустого локального репо (создает скрытую папку git)

git remote -v - (посмотреть привязку к удаленному репо)

git remote add origin 'SSH or HTTPS' - (привязка к определенному репо)

git remote remove origin - (удаление привязки к репозиторию)

git status - посмотреть статус файлов (отслеживаются или нет)

\\git add .  - запаковка//

git commit -m 'то что изменили' - маркировка коммита (описание должно быть кратким и содержательным)

git push origin branch_name - отправка данных на определенную ветку

git pull origin branch_name - стянуть данные с определенной ветки

git stash - спрятать локальные изменения (обычно применяются перед стягиванием изменений с другой ветки)

git stash apply - вернуть спрятанные ранее изменения (добавить свои изменения поверх изменений, которые стянули с другой ветки)

git branch -v - посмотреть на какой ветке находимся сейчас

git branch branch_name - создание новой ветки

git checkout branch_name - переключение на другую ветку

git branch --delete branch_name - удаление ветки
